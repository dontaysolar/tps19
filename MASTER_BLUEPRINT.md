# Kelly Formula

kelly = (win_rate * b - (1 - win_rate)) / b

# Apply limits

min_limit = 0.02           # 2% min

final_position = min(max_limit, max(min_limit, sized_position))
# = 10% (capped at max)

    final_position *= (1 - current_drawdown)
    # At 8% drawdown: 10% * 0.92 = 9.2%
